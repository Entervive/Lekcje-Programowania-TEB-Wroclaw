ZAD 6

SELECT
	r.customer_id,
    	COUNT(r.rental_id) AS "Liczba_Filmow"
FROM
	rental r
GROUP BY
	r.customer_id
ORDER BY
	Liczba_Filmow DESC
LIMIT
	10
